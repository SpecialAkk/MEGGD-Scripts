Ability Wars

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer
local teleportDelay = 0.1

local candyObjects = {}
local lastScanTime = 0

local function isCandyObject(object)
    return object.Name == "Candy Corn" or object.Name == "Candy Bowl"
end

local function scanWorkspace()
    candyObjects = {}
    
    local children = workspace:GetChildren()
    
    for i = 1, #children do
        local object = children[i]
        
        if object:IsA("Model") and isCandyObject(object) then
            local primaryPart = object.PrimaryPart or object:FindFirstChildWhichIsA("BasePart")
            if primaryPart then
                table.insert(candyObjects, primaryPart)
            end
        end
    end
end

scanWorkspace()

workspace.ChildAdded:Connect(function(object)
    if object:IsA("Model") and isCandyObject(object) then
        task.wait(0.05)
        local primaryPart = object.PrimaryPart or object:FindFirstChildWhichIsA("BasePart")
        if primaryPart then
            table.insert(candyObjects, primaryPart)
        end
    end
end)

local function teleportAllCandyObjects()
    local character = player.Character
    if not character then
        return
    end
    
    local humanoidRootPart = character:FindFirstChild("HumanoidRootPart")
    if not humanoidRootPart then
        return
    end
    
    local targetPosition = humanoidRootPart.Position
    local currentTime = tick()
    
    if currentTime - lastScanTime >= 0.3 then
        lastScanTime = currentTime
        scanWorkspace()
    end
    
    for i = 1, #candyObjects do
        local object = candyObjects[i]
        if object and object.Parent then
            object.Position = targetPosition
        end
    end
end

while true do
    teleportAllCandyObjects()
    task.wait(teleportDelay)
end
